Example of an FLTK program which uses the Bug API

Program running:

Class definition that describes the API

#include <iostream>

using namespace std;

const int NORTH = 0;
const int EAST = 1;
const int SOUTH = 2;
const int WEST = 3;
const int DIRECTIONS = 4;

class Bug
{
public:
   void move_one_unit();
   void turn_right();
   int get_row() const;
   int get_column() const;

private:
   int row;
   int column;
   int direction; // 0 = North, 1 = East, 2 = South, 3 = West
};

void Bug::move_one_unit()
{
   if (direction == NORTH) { row--; }
   else if (direction == EAST) { column++; }
   else if (direction == SOUTH) { row++; }
   else if (direction == WEST) { column--; }
}

void Bug::turn_right()
{
   direction++;
   if (direction == DIRECTIONS) { direction = 0; }
}

int Bug::get_row() const { return row; }
int Bug::get_column() const { return column; }

Create a callback to react to the user input in the "move" button in the FLUID designer

#include "bug.h"
#include "Bug.h"
static Bug bug;

Fl_Box *b=(Fl_Box *)0;

#include <FL/Fl_Image.H>
static const unsigned char idata_dog[] =
{255,255,255,255,255,255,255,255,255,255,255,255,255,255,255,255,255,255,

(lot’s of image data in this spot)

static Fl_Image *image_dog() {
  static Fl_Image *image = new Fl_RGB_Image(idata_dog, 50, 50, 3, 0);
  return image;
}

static void cb_move(Fl_Button*, void*) {
  bug.move_one_unit();
b->position(bug.get_row(),bug.get_column());
b->parent()->redraw();
}

static void cb_turn(Fl_Button*, void*) {
  bug.turn_right();
}

int main(int argc, char **argv) {
  Fl_Double_Window* w;
  { // window to contain buttons and box with bug image
    Fl_Double_Window* o = new Fl_Double_Window(350, 270, "Bug ");
    w = o; if (w) {/* empty */}
    { // box to hold image of bug
      b = new Fl_Box(25, 25, 32, 16);
      b->image( image_dog() );
    } // Fl_Box* b
    { /*
       move bug in current direction
       changle position of box which holds bug
       redraw window which is parent of box
       */
      Fl_Button* o = new Fl_Button(70, 80, 70, 20, "move");
      o->callback((Fl_Callback*)cb_move);
    } // Fl_Button* o
    { // Change current direction of bug
      Fl_Button* o = new Fl_Button(150, 80, 70, 20, "turn");
      o->callback((Fl_Callback*)cb_turn);
    } // Fl_Button* o
    o->end();
  } // Fl_Double_Window* o
  w->show(argc, argv);
  return Fl::run();
}
